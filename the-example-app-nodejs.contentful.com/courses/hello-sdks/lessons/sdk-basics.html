SDK basics

To make communication with Contentful as simple as possible, we've created open source SDKs.

Install an SDK:

cURL.NETJavaScriptJavaAndroidPHPPythonRubySwift
# N/A
dotnet add package contentful.csharp
npm install contentful
compile 'com.contentful.java:java-sdk:+'
compile 'com.contentful.java:java-sdk:+'
composer require contentful/contentful
pip install contentful
gem install contentful
// Add to your Podfile

pod 'Contentful', '~> 0.10.1'

// Then execute "pod install" in your projects directory

Include the SDK in your project and initialize a client:

  1. Add the import statement
  2. Configure a client with your space ID and Content Delivery API token
cURL.NETJavaScriptJavaAndroidPHPPythonRubySwift
export CDA_TOKEN='<access_token>';
export SPACE_ID='<space_id>';
var httpClient = new HttpClient();
var client = new ContentfulClient(httpClient, "<access_token>", "", "<space_id>");
const contentful = require('contentful')
const client = contentful.createClient({
  space: '<space_id>',
  accessToken: '<access_token>'
})

final CDAClient client =
    CDAClient
        .builder()
        .setToken("<access_token>")
        .setSpace("<space_id>")
        .build();

final CDAClient client =
    CDAClient
        .builder()
        .setToken("<access_token>")
        .setSpace("<space_id>")
        .build();

$client = new \Contentful\Delivery\Client(
    '<access_token>',
    '<space_id>'
);
import contentful

client = contentful.Client('<space_id>', '<access_token>')

require ‘contentful’

client = Contentful::Client.new(
  space: '<space_id>',
  access_token: '<access_token>',
  dynamic_entries: :auto,
  raise_errors: true
)
import Contentful

let client = Client(spaceId: "<space_id>", 
                    accessToken: "<access_token>")
